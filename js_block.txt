async function askAgent(){
  const input=document.getElementById('agentInput');
  const chat=document.getElementById('agentChat');
  const q=input.value.trim();
  if(!q)return;
  input.value='';
  // Clear the static example on first real query
  if(!window._agentUsed){chat.innerHTML='';window._agentUsed=true;}
  chat.innerHTML+=`<div style="margin:8px 0;padding:8px 12px;background:var(--s2);border-radius:8px;color:var(--t);font-family:var(--sans)">${q}</div>`;
  chat.innerHTML+=`<div id="agentThinking" style="margin:8px 0;color:var(--t4);font-size:0.72rem;font-family:var(--mono)"><span class="live-dot" style="display:inline-block;width:6px;height:6px;margin-right:6px;vertical-align:middle"></span>Running... selecting tools and executing</div>`;
  chat.scrollTop=chat.scrollHeight;
  try{
    const controller=new AbortController();
    const timeout=setTimeout(()=>controller.abort(),45000);
    const r=await fetch(`${B}/v1/ask`,{method:'POST',headers:{'Authorization':`Bearer ${K}`,'Content-Type':'application/json'},body:JSON.stringify({question:q}),signal:controller.signal});
    clearTimeout(timeout);
    const d=await r.json();
    const el=document.getElementById('agentThinking');
    if(el)el.remove();
    const tools=d.sources?d.sources.map(s=>`<span style="background:var(--rg);color:var(--r);padding:1px 6px;border-radius:3px;font-size:0.65rem;font-family:var(--mono)">${s}</span>`).join(' '):'';
    chat.innerHTML+=`<div style="margin:8px 0;padding:12px 14px;background:rgba(255,51,51,0.04);border:1px solid rgba(255,51,51,0.1);border-radius:8px">
      <div style="color:var(--t);font-family:var(--sans);font-size:0.82rem;line-height:1.7;white-space:pre-wrap">${d.answer||'Could not compute an answer.'}</div>
      <div style="margin-top:8px;display:flex;gap:6px;flex-wrap:wrap;align-items:center">
        ${tools}
        <span style="margin-left:auto;font-size:0.6rem;color:var(--t4);font-family:var(--mono)">${d.credits||'?'}cr &middot; ${d.time_ms||'?'}ms &middot; real computation</span>
      </div>
    </div>`;
    chat.scrollTop=chat.scrollHeight;
  }catch(e){
    const el=document.getElementById('agentThinking');
    if(el){
      if(e.name==='AbortError'){
        el.innerHTML='<span style="color:var(--o)">Server is warming up (cold start). Try again in a few seconds, or use the tool picker below.</span>';
      } else {
        el.innerHTML='<span style="color:var(--o)">Connecting to server... try again in a moment or use the tool picker below.</span>';
      }
    }
  }
}

async function init(){
  // Always load embedded catalog first (works without server)
  loadCatalog(EMBEDDED);
  // Pre-warm the server (Railway cold start takes 10-30s, start it early)
  fetch(`${B}/v1/health`).catch(()=>{});
  // Then try to upgrade from live server
  try{
    const r=await fetch(`${B}/v1/tools?limit=500`);const d=await r.json();
    const live=d.apis||d.tools||[];
    if(live.length>0)loadCatalog(live);
  }catch(e){/* server not available, embedded catalog already loaded */}
}
