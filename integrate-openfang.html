Rust Agent OS

Integrate Slopshop with OpenFang

OpenFang is a Rust-based agent OS built for low-overhead, high-concurrency workloads. Add Slopshop persistent verifiable memory via simple async HTTP from reqwest — no special SDK, just two crates.

Prerequisites
Rust 1.70+, an existing OpenFang agent project, Tokio async runtime, and a free Slopshop account. Get your API key with slop status after installing the CLI.
1Add Dependencies to Cargo.toml

Add two crates to your Cargo.toml. reqwest handles HTTP with JSON support, and serde_json handles value serialization. Both are widely used and compile quickly.

Cargo.toml
[dependencies]
reqwest = { version = "0.11", features = ["json"] }
serde_json = "1"
tokio = { version = "1", features = ["full"] }  # if not already present

Then run cargo build to fetch and compile dependencies.

2The SlopshopClient

Create a new file src/slop.rs in your OpenFang project with this reusable async client. It wraps all three core operations: remember, recall, and search.

src/slop.rs
use reqwest::Client;
use serde_json::{json, Value};

const SLOP_KEY: &str = "sk-slop-your-key-here";
const SLOP_BASE: &str = "https://slopshop.gg";

pub struct SlopshopClient {
    client: Client,
    api_key: String,
}

impl SlopshopClient {
    pub fn new(api_key: impl Into<String>) -> Self {
        Self { client: Client::new(), api_key: api_key.into() }
    }

    pub async fn remember(&self, key: &str, value: Value, namespace: &str)
        -> Result<Value, reqwest::Error>
    {
        self.client
            .post(format!("{}/v1/memory-set", SLOP_BASE))
            .bearer_auth(&self.api_key)
            .json(&json!({"key": key, "value": value, "namespace": namespace}))
            .send().await?
            .json().await
    }

    pub async fn recall(&self, key: &str, namespace: &str)
        -> Result<Value, reqwest::Error>
    {
        self.client
            .post(format!("{}/v1/memory-get", SLOP_BASE))
            .bearer_auth(&self.api_key)
            .json(&json!({"key": key, "namespace": namespace}))
            .send().await?
            .json().await
    }

    pub async fn search(&self, query: &str, namespace: &str, limit: u32)
        -> Result<Value, reqwest::Error>
    {
        self.client
            .post(format!("{}/v1/memory-search", SLOP_BASE))
            .bearer_auth(&self.api_key)
            .json(&json!({"query": query, "namespace": namespace, "limit": limit}))
            .send().await?
            .json().await
    }

    pub async fn vault_proxy(
        &self,
        vault_id: &str,
        url: &str,
        method: &str,
        body: Option<Value>
    ) -> Result<Value, reqwest::Error>
    {
        self.client
            .post(format!("{}/v1/vault/proxy", SLOP_BASE))
            .bearer_auth(&self.api_key)
            .json(&json!({
                "vault_id": vault_id,
                "url": url,
                "method": method,
                "body": body
            }))
            .send().await?
            .json().await
    }
}
Production tip
Load the API key from an environment variable rather than a constant: std::env::var("SLOP_API_KEY").expect("SLOP_API_KEY not set"). Pass it into SlopshopClient::new() at startup.
3Usage in an OpenFang Agent

Wire the client into your OpenFang agent's task loop. The example below shows an analysis agent that recalls prior context, runs its task, then stores findings with a verifiable proof.

src/agents/analyst.rs
use crate::slop::SlopshopClient;
use serde_json::json;

pub struct AnalystAgent {
    slop: SlopshopClient,
    namespace: String,
}

impl AnalystAgent {
    pub fn new(api_key: &str) -> Self {
        Self {
            slop: SlopshopClient::new(api_key),
            namespace: "openfang-analyst".to_string(),
        }
    }

    pub async fn run(&self, task: &str) -> anyhow::Result<()> {
        // Recall shared context from prior agents
        let context = self.slop
            .recall("workflow/context", &self.namespace)
            .await?;

        println!("Prior context: {}", context["value"]);

        // Simulate analysis work
        let findings = json!({
            "task": task,
            "result": "analysis complete",
            "confidence": 0.97,
            "agent": "analyst-v1"
        });

        // Persist with verifiable proof
        let proof = self.slop
            .remember("analyst/findings", findings, &self.namespace)
            .await?;

        println!(
            "Stored findings | proof_hash={}...",
            &proof["proof_hash"]
                .as_str()
                .unwrap_or("")
                .chars()
                .take(24)
                .collect::<String>()
        );

        Ok(())
    }
}

#[tokio::main]
async fn main() {
    let api_key = std::env::var("SLOP_API_KEY").expect("SLOP_API_KEY not set");
    let agent = AnalystAgent::new(&api_key);
    agent.run("Analyze Q1 market data").await.unwrap();
}
4Error Handling

For production OpenFang agents, wrap the client methods with typed error handling. The reqwest::Error type covers network failures, timeouts, and non-2xx responses with JSON body parsing.

src/slop.rs — with error handling
use thiserror::Error;

#[derive(Error, Debug)]
pub enum SlopError {
    #[error("HTTP error: {0}")]
    Http(#[from] reqwest::Error),
    #[error("API error: {status} — {message}")]
    Api { status: String, message: String },
}

impl SlopshopClient {
    pub async fn remember_checked(
        &self,
        key: &str,
        value: serde_json::Value,
        namespace: &str
    ) -> Result<String, SlopError> {
        let resp = self.remember(key, value, namespace).await?;

        if resp["status"] == "stored" {
            Ok(resp["proof_hash"]
                .as_str()
                .unwrap_or("")
                .to_string())
        } else {
            Err(SlopError::Api {
                status: resp["status"].to_string(),
                message: resp["error"].as_str().unwrap_or("unknown").to_string(),
            })
        }
    }
}
