quickstart
# Store a memory
curl -X POST /v1/memory/store \
  -H "Authorization: Bearer $SLOP_KEY" \
  -d '{"content":"User prefers concise answers. Expert in Rust."}'
 
# → { "id": "mem_9a3z1", "stored": true }
 
# Run Dream Engine overnight
curl -X POST /v1/memory/dream/start \
  -d '{"strategy":"synthesize,compress,insight_generate"}'
 
# → { "dream_id": "drm_7f2b", "stages": 9, "status": "running" }
